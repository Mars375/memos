    1. ADD import at the top of `create_fastapi_app` (after other imports inside the function), or at module level:

    2. ADD exception handler AFTER `app = FastAPI(...)` and BEFORE middleware registration:
    ```python
    @app.exception_handler(RequestValidationError)
    async def _validation_exception_handler(request, exc: RequestValidationError):
        """Wrap Pydantic errors in MemOS standard error format for backward compat."""
        errors = exc.errors()
        if len(errors) == 1:
            e = errors[0]
            field = " -> ".join(str(loc) for loc in e.get("loc", []) if loc != "body")
            msg = f"{field}: {e['msg']}" if field else e["msg"]
        else:
            parts = []
            for e in errors:
                field = " -> ".join(str(loc) for loc in e.get("loc", []) if loc != "body")
                parts.append(f"{field}: {e['msg']}" if field else e["msg"])
            msg = "; ".join(parts)
        from fastapi.responses import JSONResponse
        return JSONResponse(
            status_code=422,
            content={"status": "error", "message": msg, "code": "VALIDATION"},
        )
    ```

    This ensures that when Pydantic rejects a payload (e.g., missing `subject` in FactRequest), the client gets `{"status": "error", "message": "subject: Field required", "code": "VALIDATION"}` instead of the raw FastAPI `{"detail": [...]}` format.
